# comfy_wan22_gui.py v0.9 → v0.10 改动清单

本文件记录从 v0.9 升级到 v0.10 的所有代码改动。改动主题:
1. **每张首尾帧生成结束后,自动新开 GPT 对话** —— 防止上下文污染、画风漂移、参考图串味。
2. **上传参考图后校验附件数量** —— 如果输入框里的实际附件数 ≠ 期望数,说明有残留,自动新开对话并重试一次。

所有行号以仓库当前 `main` 分支 commit 为准(共 4842 行)。改动按出现顺序排列。

## 改动 2:给 `GPTWebController` 加两个新方法

**位置**:第 1531 行 `upload_files` 方法结尾的 `return valid_paths` 之后、`# ============== GUI ==============` 之前。

**插入以下两个新方法**:

```python
    def new_chat(self, verify_timeout=5):
        """
        开启一个新的 GPT 对话。策略按优先级:
          1. 找「新对话 / New chat」按钮点击
          2. 找不到就 navigate 回基础 URL(大部分 GPT 镜像站等价于新对话)
        成功后会尝试等待输入框 #prompt-textarea 重新出现,确认新对话已就绪。
        """
        if not self.is_alive():
            raise RuntimeError("浏览器未启动")
        d = self.driver
        self.log("🔄 开启新对话 ...")
        clicked = False
        # 方案 A:点「新对话」按钮
        try:
            clicked_sel = d.execute_script("""
                const selectors = [
                    'a[href="/"]',                                // ChatGPT 官方
                    'a[data-testid="create-new-chat-button"]',
                    'button[data-testid="create-new-chat-button"]',
                    'a[aria-label*="New chat" i]',
                    'a[aria-label*="新对话" i]',
                    'a[aria-label*="新聊天" i]',
                    'button[aria-label*="New chat" i]',
                    'button[aria-label*="新对话" i]',
                    'button[aria-label*="新聊天" i]',
                ];
                for (const s of selectors) {
                    const el = document.querySelector(s);
                    if (el) { el.click(); return s; }
                }
                return null;
            """)
            if clicked_sel:
                self.log(f"   ✓ 已点击新对话按钮: {clicked_sel}")
                clicked = True
        except Exception as e:
            self.log(f"   ⚠️ 点击按钮失败: {e}")

        # 方案 B:回滚到基础 URL
        if not clicked:
            try:
                d.get(self.url)
                self.log(f"   ✓ 已 navigate 回: {self.url}")
            except Exception as e:
                self.log(f"   ❌ navigate 失败: {e}")
                return False

        # 等输入框就绪
        try:
            WebDriverWait(d, verify_timeout).until(
                EC.presence_of_element_located((By.CSS_SELECTOR, self.INPUT_SELECTOR))
            )
            self.log("   ✅ 新对话已就绪")
            return True
        except Exception:
            self.log("   ⚠️ 新对话输入框未检测到,但可能已开(继续)")
            return True

    def count_composer_attachments(self):
        """
        统计当前输入框区域里已挂载的附件(图片)数量。
        用于"上传后应有 N 张,实际 >N 说明有残留"这类一致性校验。
        返回 int。
        """
        if not self.is_alive():
            return 0
        try:
            n = self.driver.execute_script("""
                // 尽量覆盖主流 GPT 镜像站的附件缩略图结构
                const sels = [
                    'div[class*="attachment" i] img',
                    'div[class*="file-preview" i] img',
                    'img[alt*="附件" i]',
                    'div[role="presentation"] img',
                    'div[class*="Thumbnail" i] img',
                    'button[aria-label*="remove" i][aria-label*="attachment" i]',
                ];
                let set = new Set();
                for (const s of sels) {
                    document.querySelectorAll(s).forEach(el => set.add(el));
                }
                // 只统计在输入框附近的(过滤掉对话流里回复区的图)
                // 简化策略:取视口下半部分的,通常 composer 在底部
                let count = 0;
                const vh = window.innerHeight;
                set.forEach(el => {
                    try {
                        const r = el.getBoundingClientRect();
                        if (r.top > vh * 0.4) count++;
                    } catch (_) {}
                });
                return count;
            """)
            return int(n or 0)
        except Exception:
            return 0

```

```python
    def _batch_gpt_images(self, save_dir, only_missing):
        Path(save_dir).mkdir(parents=True, exist_ok=True)
        # 确保素材库索引已建
        if not self.asset_lib._index:
            d = self.asset_dir_var.get().strip()
            if d and Path(d).exists():
                try:
                    self.asset_lib.set_root(d)
                except Exception:
                    pass

        total = len(self.scenes)
        done = 0
        self._log(f"\n🖼 开始批量 GPT 生图 v0.10 两次调用 + 每场景新对话模式"
                  f"(共 {total} 个场景,目标目录:{save_dir})")

        for i, s in enumerate(self.scenes):
            if self.stop_flag.is_set():
                self._log("⏹ 已停止")
                break
            while self.gpt_pause_flag.is_set():
                if self.stop_flag.is_set():
                    break
                time.sleep(0.5)
            if self.stop_flag.is_set():
                self._log("⏹ 已停止")
                break

            if only_missing and s.get("start") and s.get("end"):
                continue

            self._log(f"\n🎨 [{i+1}/{total}] {s['name']}")

            # v0.10: 每个场景开始前,先新开一个 GPT 对话
            try:
                self.gpt_ctrl.new_chat()
                time.sleep(1.5)
            except Exception as e:
                self._log(f"   ⚠️ new_chat 失败(继续用当前对话):{e}")

            try:
                # ─── 1. 组装基础 prompt ───
                raw = (s.get("gpt_prompt") or "").strip() or (s.get("prompt") or "").strip()
                if not raw:
                    self._log("   ⚠️ 无 prompt,跳过")
                    continue

                cid = s.get("color", "")
                if cid and self.auto_inject_color_var.get():
                    raw = self.color_anchors.inject(raw, cid)
                    self._log(f"   🎨 已注入色板:{self.color_anchors.get_name(cid)}")

                has_people = any(kw in raw for kw in self._PEOPLE_KEYWORDS)
                self._log(f"   🧭 prompt 模式:{'人物镜头' if has_people else '纯环境空镜'}")

                # ─── 2. 匹配素材库 + 上传 + 附件数校验 ───
                matched = self.asset_lib.match(raw) if self.asset_lib._index else []
                ref_paths = [p for _, p in matched]
                if ref_paths and self.auto_upload_var.get():
                    expected = len(ref_paths)
                    self._log(f"   📎 匹配到 {expected} 张参考图:"
                              f"{', '.join(k for k, _ in matched)}")
                    # v0.10: 上传 + 校验;不对就重开一次
                    ok = self._upload_with_verify(ref_paths, expected)
                    if not ok:
                        self._log("   ⚠️ 附件数校验失败且重试后仍不一致,跳过本场景")
                        continue

                sub = Path(save_dir) / f"{i+1:02d}_{s['name']}"

                # ─── 3. 第一次请求:首帧 ───
                self._log(f"   📤 [阶段 1/2] 请求首帧图...")
                start_prompt = self._build_start_frame_prompt(raw, has_people)
                start_urls, _ = self._request_single_gpt_image(
                    start_prompt, timeout=300, stable_seconds=5)
                if not start_urls:
                    self._log("   ❌ 首帧超时未拿到(检查 GPT / 镜像站 DOM)")
                    continue
                self._log(f"   📥 首帧拿到 {len(start_urls)} 张")

                start_saved = self.gpt_ctrl.download_last_images(
                    str(sub), urls=start_urls[:1])
                if not start_saved:
                    self._log("   ❌ 首帧图下载失败")
                    continue
                start_path = start_saved[0]
                self._log(f"   ✅ 首帧已保存: {start_path}")

                time.sleep(2)
                if self.stop_flag.is_set():
                    self._log("⏹ 已停止(首帧完成后)")
                    if not only_missing or not s.get("start"):
                        s["start"] = start_path
                    self.scenes[i] = s
                    self.root.after(0, self._refresh_tree)
                    break

                while self.gpt_pause_flag.is_set():
                    if self.stop_flag.is_set():
                        break
                    time.sleep(0.5)

                # ─── 4. 第二次请求:尾帧(和首帧同对话,保持一致性) ───
                self._log(f"   📤 [阶段 2/2] 请求尾帧图(同对话,基于首帧延续)...")
                end_prompt = self._build_end_frame_prompt(raw, has_people)
                end_urls, _ = self._request_single_gpt_image(
                    end_prompt, timeout=300, stable_seconds=5)
                if not end_urls:
                    self._log("   ⚠️ 尾帧超时,回退为用首帧作为尾帧(可手动替换)")
                    end_path = start_path
                else:
                    self._log(f"   📥 尾帧拿到 {len(end_urls)} 张")
                    end_saved = self.gpt_ctrl.download_last_images(
                        str(sub), urls=end_urls[:1])
                    if not end_saved:
                        self._log("   ⚠️ 尾帧下载失败,回退用首帧")
                        end_path = start_path
                    else:
                        end_path = end_saved[0]
                        self._log(f"   ✅ 尾帧已保存: {end_path}")

                # ─── 5. 回填 ───
                if only_missing:
                    if not s.get("start"):
                        s["start"] = start_path
                    if not s.get("end"):
                        s["end"] = end_path
                else:
                    s["start"] = start_path
                    s["end"] = end_path
                self.scenes[i] = s

                self.root.after(0, self._refresh_tree)
                done += 1

                self.progress["value"] = i + 1
                self.progress_label.config(text=f"{i+1} / {total}")

                time.sleep(3)

            except Exception as e:
                self._log(f"   ❌ 失败: {e}")

        self._log(f"\n🎉 批量 GPT 生图完成!成功 {done} 个")
        self.gpt_batch_running = False
        self.root.after(0, self._update_gpt_batch_buttons)
        self._save_config()
```

## 改动 4:新增 `_upload_with_verify` 辅助方法

**位置**:紧跟改动 3 的 `_batch_gpt_images` 方法之后插入。

```python
    # ---------- v0.10: 上传参考图 + 数量校验 + 不对就重来 ----------
    def _upload_with_verify(self, ref_paths, expected, max_retry=1):
        """
        上传参考图,上传完后用 count_composer_attachments 校验。
        数量对不上 = 有残留或丢失,新开对话重试一次。

        返回 True 表示最终成功(或没 selenium 可校验但文件已发出去),
             False 表示重试仍不一致。
        """
        for attempt in range(max_retry + 1):
            try:
                self.gpt_ctrl.upload_files(ref_paths)
                time.sleep(1.8)
            except Exception as e:
                self._log(f"   ⚠️ 参考图上传失败(第 {attempt+1} 次):{e}")
                if attempt >= max_retry:
                    return False
                try:
                    self.gpt_ctrl.new_chat()
                    time.sleep(1.2)
                except Exception:
                    pass
                continue

            # 校验附件数
            actual = self.gpt_ctrl.count_composer_attachments()
            if actual == expected:
                self._log(f"   ✅ 附件数校验通过:{actual}/{expected}")
                return True
            elif actual == 0:
                # 检测不到(镜像站 DOM 不支持)不拦,直接放过
                self._log(f"   ℹ 未能检测到附件缩略图,跳过校验(DOM 不匹配)")
                return True
            else:
                self._log(
                    f"   ⚠️ 附件数对不上:期望 {expected} 张,实际 {actual} 张"
                    f" —— 多半是上一场景残留,新开对话重试(第 {attempt+1} 次)"
                )
                if attempt >= max_retry:
                    return False
                try:
                    self.gpt_ctrl.new_chat()
                    time.sleep(1.2)
                except Exception as e:
                    self._log(f"   ⚠️ new_chat 失败:{e}")
                    return False
        return False
```
